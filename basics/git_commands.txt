sudo app install git - установка системы контроля версий

git --version - проверка версии

git config --global user.email "your@email.com"
git --global user.name "yourName"

git config --list - проверка настроек

ssh-kaygen -  создали ssh  ключ

cat ~/.ssh/id_rsa.pub - получение ssh  ключа

создаем репозитори на github 

переходим в папку с нашим проектом

git init -  превращает текущую директорию в git репозиторий

git remote add название связи (обычно origin)  ссылку на удаленный репозитори ssh

git remote -v - просмотр списка удаленный репозиториев

git remote rm название связи (обычно origin) удаление связи с репозторием.

git status - проверка статуса файлов

git add название файлов - отслеживание конкретный файлов
git add. - отслеживание всех файлов

git commit - создание коммита (версии слепка кода)
git commit -m "описание версии" - создание коммита с комментариями

 git push название связи (обычно - origin) 
 название ветки - отправка версии кода на удаленный репозиторий
 Например: git push origin master 

 git clone ссылка на кдаленный репозиторий - склонировать либо скопировать либо скачатьудаленный репозиторий

 git diff - просмотр внесенных изменений

 git brench - вывод списка веток в терминал (звездочкой помечается активная ветка)

 git brench название ветки - создание указанной ветки

 git chechout название ветки
 git switch название ветки - переход на указанную ветку


git pull название связи название ветки - стягивание изменений с удаленного репозитория, с указанной ветки.

git stash  - спрятать локальные данные во временную область.

git stash list -  просмотр списка спрятанных данных

git stash apply -  запись спрятанных данных.

git stash drop -  удаление спрятанных данных.

git log -  просмотр коммитов, версий.
git log --oneline - компактный вид лога.
